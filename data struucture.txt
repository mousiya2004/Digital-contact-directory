<script>
let currentUser=null,contacts=[],editIndex=-1;

function toast(msg,error=false){
  const t=document.getElementById('toast');
  document.getElementById('toastMsg').textContent=msg;
  document.getElementById('toastIcon').textContent=error?'✕':'✓';
  t.classList.toggle('error',error);
  t.classList.add('show');
  setTimeout(()=>t.classList.remove('show'),2800);
}

function register(){
  const u=document.getElementById('username').value.trim(),p=document.getElementById('password').value.trim();
  if(!u||!p)return toast('Please fill in both fields',true);
  if(localStorage.getItem('user_'+u))return toast('Username already taken',true);
  localStorage.setItem('user_'+u,p);
  toast('Account created! You can now log in.');
}

function login(){
  const u=document.getElementById('username').value.trim(),p=document.getElementById('password').value.trim();
  if(!u||!p)return toast('Enter your credentials',true);
  const stored=localStorage.getItem('user_'+u);
  if(stored===p){currentUser=u;startApp();}
  else{toast('Incorrect username or password',true);}
}

function logout(){currentUser=null;location.reload();}

function startApp(){
  document.getElementById('authBox').classList.add('hidden');
  document.getElementById('appBox').classList.remove('hidden');
  document.getElementById('welcome').textContent=currentUser;
  document.getElementById('avatarInitial').textContent=currentUser[0].toUpperCase();
  load();render();
}

function load(){contacts=JSON.parse(localStorage.getItem('contacts_'+currentUser)||'[]');}
function save(){localStorage.setItem('contacts_'+currentUser,JSON.stringify(contacts));}

function submitContact(){
  const n=document.getElementById('cName').value.trim(),ph=document.getElementById('cPhone').value.trim(),em=document.getElementById('cEmail').value.trim();
  if(!n||!ph||!em)return toast('Please fill in all fields',true);
  if(editIndex>=0){
    contacts[editIndex]={name:n,phone:ph,email:em,added:contacts[editIndex].added};
    toast('Contact updated');cancelEdit();
  }else{
    contacts.push({name:n,phone:ph,email:em,added:Date.now()});
    toast('Contact added');
    document.getElementById('cName').value='';
    document.getElementById('cPhone').value='';
    document.getElementById('cEmail').value='';
  }
  save();render();
}

function editContact(i){
  const c=contacts[i];
  document.getElementById('cName').value=c.name;
  document.getElementById('cPhone').value=c.phone;
  document.getElementById('cEmail').value=c.email;
  editIndex=i;
  document.getElementById('submitBtn').textContent='✓ Update Contact';
  document.getElementById('editBadge').classList.remove('hidden');
  document.getElementById('cancelBtn').classList.remove('hidden');
  document.getElementById('formCard').classList.add('editing');
  document.getElementById('cName').focus();
}

function cancelEdit(){
  editIndex=-1;
  document.getElementById('cName').value='';
  document.getElementById('cPhone').value='';
  document.getElementById('cEmail').value='';
  document.getElementById('submitBtn').textContent='＋ Add Contact';
  document.getElementById('editBadge').classList.add('hidden');
  document.getElementById('cancelBtn').classList.add('hidden');
  document.getElementById('formCard').classList.remove('editing');
}

let currentSort='name';
function setSort(type){
  currentSort=type;
  document.querySelectorAll('.sort-btn').forEach(b=>b.classList.remove('active'));
  document.getElementById('sort'+type.charAt(0).toUpperCase()+type.slice(1)).classList.add('active');
  render();
}

let deleteTargetIndex=-1;
function deleteContact(i){
  deleteTargetIndex=i;
  document.getElementById('modalName').textContent='"'+contacts[i].name+'"';
  document.getElementById('deleteModal').classList.add('show');
}
function closeDeleteModal(){
  document.getElementById('deleteModal').classList.remove('show');
  deleteTargetIndex=-1;
}
function confirmDelete(){
  if(deleteTargetIndex<0)return;
  const name=contacts[deleteTargetIndex].name,cardId='card-'+deleteTargetIndex;
  closeDeleteModal();
  const card=document.getElementById(cardId);
  if(card){
    card.classList.add('deleting');
    setTimeout(()=>{contacts.splice(deleteTargetIndex,1);deleteTargetIndex=-1;save();render();toast(name+' removed');},340);
  }else{
    contacts.splice(deleteTargetIndex,1);deleteTargetIndex=-1;save();render();toast(name+' removed');
  }
}

function render(){
  const q=document.getElementById('search').value.toLowerCase();
  const filtered=contacts.filter(c=>
    c.name.toLowerCase().includes(q)||
    c.phone.toLowerCase().includes(q)||
    c.email.toLowerCase().includes(q)
  );
  document.getElementById('statTotal').textContent=contacts.length;
  document.getElementById('statShowing').textContent=filtered.length;
  const list=document.getElementById('list');
  if(filtered.length===0){
    list.innerHTML=`<div class="empty-state"><div class="icon">${contacts.length===0?'📋':'🔎'}</div><p>${contacts.length===0?'No contacts yet. Add your first one above.':'No contacts match your search.'}</p></div>`;
    return;
  }
  let sorted=[...filtered];
  if(currentSort==='name')sorted.sort((a,b)=>a.name.localeCompare(b.name));
  else if(currentSort==='nameDesc')sorted.sort((a,b)=>b.name.localeCompare(a.name));
  else if(currentSort==='newest')sorted.sort((a,b)=>(b.added||0)-(a.added||0));
  else if(currentSort==='oldest')sorted.sort((a,b)=>(a.added||0)-(b.added||0));
  list.innerHTML=sorted.map((c,i)=>{
    const realIndex=contacts.indexOf(c);
    const initials=c.name.split(' ').map(w=>w[0]).join('').slice(0,2).toUpperCase();
    const colors=['#5b45e0,#9b80ff','#e8453c,#ff8c5a','#00a882,#00c4ff','#d4921a,#e8b84b'];
    const grad=colors[i%colors.length];
    return`<div class="contact-card" id="card-${realIndex}">
      <div class="contact-avatar" style="background:linear-gradient(135deg,${grad})">${initials}</div>
      <div class="contact-info">
        <div class="contact-name">${c.name}</div>
        <div class="contact-detail">${c.phone} · ${c.email}</div>
      </div>
      <div class="contact-actions">
        <button class="action-btn edit" onclick="editContact(${realIndex})" title="Edit">✎</button>
        <button class="action-btn del" onclick="deleteContact(${realIndex})" title="Delete">🗑</button>
      </div>
    </div>`;
  }).join('');
}
